# Здесь priority указывается в sitemap.xml, robots.txt только разрешает

# Отзывы — средний приоритет

# Можно указать, что сканировать реже

# Чат — низкий приоритет, индексировать необязательно, но разрешаем
Allow: /chat/

# Указание на карту сайта для улучшенной индексации
